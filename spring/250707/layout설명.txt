##컨트롤러를 다음과 같이 수정
@RequestMapping("index")
	public String index(Model model) {
		model.addAttribute("page", "board/"+"index"); // views/board/index.jsp -> board/index만을 attribute에 저장
		return "index"; //무조건 views/index.jsp
}
@RequestMapping("index")
	public String index(Model model) {
		model.addAttribute("page", "login/"+ "index");
		return "index"; //무조건 views/index.jsp
}

@RequestMapping("index")
	public String index(Model model) {
		model.addAttribute("page", "member/"+ "index");
		return "index"; //무조건 views/index.jsp
}

1)layout프로젝트 생성 
2)pom.xml, dispacher-servlet.xml, web.xml
